// ── Label + icon maps ──────────────────────────────────────
var SKIN_LABELS    = { oily:'Oily', dry:'Dry', combination:'Combination', sensitive:'Sensitive' };
var CONCERN_LABELS = { acne:'Acne', glow:'Glow', darkspots:'Dark Spots', daily:'Daily Care' };
var BUDGET_LABELS  = { low:'Low (₹100–₹500)', medium:'Medium (₹500–₹2,000)', high:'High (₹2,000+)' };
var PROD_ICONS     = { 'Face Wash':'🧴', 'Moisturizer':'💧', 'Sunscreen':'☀️' };
var SKIN_ICONS     = { oily:'✨', dry:'🌿', combination:'⚖️', sensitive:'🌸' };
var CONCERN_ICONS  = { acne:'🎯', glow:'💫', darkspots:'🔬', daily:'🌅' };
var PROD_COLORS    = { 'Face Wash':'#4f8ef7', 'Moisturizer':'#34b87a', 'Sunscreen':'#f5a623' };

// ── Run Beauty Engine ──────────────────────────────────────
function runBeautyEngine() {
  var skin    = document.getElementById('beSkin').value;
  var concern = document.getElementById('beConcern').value;
  var budget  = document.getElementById('beBudget').value;
  var result  = document.getElementById('beautyResult');
  var btn     = document.querySelector('.beauty-btn');

  if (!skin || !concern || !budget) {
    result.innerHTML = '<p class="eng-error">&#9888; Please select all 3 options to get your skincare picks.</p>';
    result.classList.add('be-visible');
    return;
  }

  var picks = BEAUTY_DATA[skin] && BEAUTY_DATA[skin][concern] && BEAUTY_DATA[skin][concern][budget];

  if (!picks) {
    result.innerHTML = '<p class="eng-error">No match found. Try a different combination.</p>';
    result.classList.add('be-visible');
    return;
  }

  btn.disabled = true;
  btn.innerHTML = '<span class="eng-spinner"></span> Finding your skincare picks\u2026';
  btn.classList.add('eng-loading');
  result.classList.remove('be-visible');
  result.innerHTML = '';

  setTimeout(function () {
    var skinLabel    = SKIN_LABELS[skin]       || skin;
    var concernLabel = CONCERN_LABELS[concern] || concern;
    var budgetLabel  = BUDGET_LABELS[budget]   || budget;

    var html  = '<div class="be-banner">';
    html += '<span class="be-banner-eyebrow">&#9733; Best Pick for You</span>';
    html += '<span class="be-banner-title">Your Complete Skincare Routine</span>';
    html += '<p class="be-banner-reason">Based on your <strong>' + skinLabel + '</strong> skin, <strong>' + concernLabel + '</strong> concern, and <strong>' + budgetLabel + '</strong> — we analysed the options and made the call. This is your routine.</p>';
    html += '</div>';

    html += '<div class="be-tags-row">';
    html += '<span class="eng-rtag">' + SKIN_ICONS[skin]       + ' ' + skinLabel    + ' Skin</span>';
    html += '<span class="eng-rtag">' + CONCERN_ICONS[concern] + ' ' + concernLabel + '</span>';
    html += '<span class="eng-rtag">&#128176; ' + budgetLabel + '</span>';
    html += '</div>';

    html += '<div class="be-cards">';
    ['facewash','moisturizer','sunscreen'].forEach(function (key, idx) {
      var p     = picks[key];
      var icon  = PROD_ICONS[p.type]   || '&#10024;';
      var color = PROD_COLORS[p.type]  || 'var(--text)';
      html += '<div class="be-card" style="animation-delay:' + (idx * 0.09) + 's">';

      html += '<div class="be-card-confirm-strip">';
      html += '<span>&#10003; Best ' + p.type + ' For You</span>';
      html += '</div>';

      html += '<div class="be-card-strip" style="background:' + color + '">';
      html += '<span>' + icon + ' ' + p.type + '</span>';
      html += '</div>';

      html += '<div class="be-card-body">';
      html += '<div class="be-card-name">' + p.name + '</div>';
      html += '<div class="be-card-price">' + p.price + '</div>';

      html += '<p class="be-card-context">For <strong>' + skinLabel + '</strong> skin with <strong>' + concernLabel + '</strong> concern:</p>';
      html += '<p class="be-card-reason">' + p.reason + '</p>';

      html += '<div class="be-final-verdict">';
      html += '<span class="be-verdict-chip">FINAL DECISION</span>';
      html += '<p>This is the best ' + p.type + ' for your ' + skinLabel.toLowerCase() + ', ' + concernLabel.toLowerCase() + '-focused skin.</p>';
      html += '</div>';

      html += '<a class="be-buy-btn" href="' + p.link + '" target="_blank" rel="noopener">Buy Now &#8594;</a>';
      html += '</div>';
      html += '</div>';
    });
    html += '</div>';
    html += '<p class="eng-disclaimer">&#128161; Use all 3 products together for the best possible results.</p>';

    result.innerHTML = html;
    btn.disabled = false;
    btn.innerHTML = '<span class="btn-icon">&#10024;</span> Get My Skincare Picks';
    btn.classList.remove('eng-loading');

    requestAnimationFrame(function () {
      requestAnimationFrame(function () {
        result.classList.add('be-visible');
        setTimeout(function () {
          result.scrollIntoView({ behavior: 'smooth', block: 'start' });
        }, 80);
      });
    });
  }, 900);
}
